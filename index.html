#!/bin/bash
# فقط این دو مقدار را تغییر دهید
TUNNEL_AUTH_KEY="YOUR_SECRET_KEY"
SSH_PORT=22

TUNNEL_PORT=8080
export DEBIAN_FRONTEND=noninteractive
docker rm -f tunnel-node mhrv-tunnel 2>/dev/null || true
docker rmi -f ghcr.io/therealaleph/mhrv-tunnel-node 2>/dev/null || true
apt update -y -q && apt upgrade -y -q
apt install -y -q docker.io curl ufw
systemctl enable docker --now
ufw --force reset
ufw default deny incoming
ufw default allow outgoing
ufw allow ${SSH_PORT}/tcp
ufw allow ${TUNNEL_PORT}/tcp
ufw --force enable
docker pull ghcr.io/therealaleph/mhrv-tunnel-node:latest
docker run -d \
  --name tunnel-node \
  --restart always \
  -p ${TUNNEL_PORT}:${TUNNEL_PORT} \
  -e TUNNEL_AUTH_KEY="${TUNNEL_AUTH_KEY}" \
  -e PORT=${TUNNEL_PORT} \
  ghcr.io/therealaleph/mhrv-tunnel-node:latest
sleep 5
curl -sf http://localhost:${TUNNEL_PORT}/health && echo " OK"
echo "آدرس VPS: http://$(curl -4s ifconfig.me):${TUNNEL_PORT}"

#!/bin/bash
# فقط رمز را تغییر دهید
TUNNEL_AUTH_KEY="YOUR_SECRET_KEY"

G='\033[0;32m'; R='\033[0;31m'; B='\033[1;34m'; NC='\033[0m'
ok(){ echo -e "  ${G}OK${NC} $1"; }
err(){ echo -e "  ${R}ERR${NC} $1"; }

echo -e "\n${B}Docker${NC}"
docker ps --format "  {{.Names}} | {{.Status}}" | grep tunnel \
  && ok "در حال اجراست" || err "اجرا نیست"

echo -e "\n${B}Health${NC}"
H=$(curl -sf --max-time 5 http://localhost:${TUNNEL_PORT}/health 2>/dev/null || echo "FAIL")
[ "$H" = "ok" ] && ok "سرویس OK" || err "پاسخ نمی‌دهد"

echo -e "\n${B}Tunnel Test${NC}"
C=$(curl -s --max-time 10 -X POST http://localhost:${TUNNEL_PORT}/tunnel \
  -H "Content-Type: application/json" \
  -d "{\"k\":\"${TUNNEL_AUTH_KEY}\",\"op\":\"connect\",\"host\":\"www.google.com\",\"port\":443}" 2>/dev/null)
echo "$C" | grep -q '"sid"' && ok "اتصال برقرار" || err "اتصال ناموفق: $C"

echo -e "\n${B}IP${NC}"
echo "  http://$(curl -4s ifconfig.me):${TUNNEL_PORT}"

echo -e "\n${B}Firewall${NC}"
ufw status | grep -q "${TUNNEL_PORT}" && ok "پورت باز" || err "پورت بسته"

echo -e "\n${B}Resources${NC}"
docker stats tunnel-node --no-stream \
  --format "  CPU: {{.CPUPerc}} | RAM: {{.MemUsage}}" 2>/dev/null

echo -e "\n${B}Logs${NC}"
docker logs tunnel-node --tail 10 2>&1 | sed 's/^/  /'

  host:1,connection:1,"content-length":1,

function doPost(e) {
  try {
    var req = JSON.parse(e.postData.contents);
    if (req.k !== AUTH_KEY) return _json({e:"unauthorized"});
    if (req.t) return _doTunnel(req);
    if (Array.isArray(req.q)) return _doBatch(req.q);
    return _doSingle(req);
  } catch(err) { return _json({e:String(err)}); }
}

function _doTunnel(req) {
  if (req.t === "batch") return _doTunnelBatch(req);
  var p = {k:TUNNEL_AUTH_KEY};
  switch(req.t) {
    case "connect": p.op="connect";p.host=req.h;p.port=req.p; break;
    case "data":    p.op="data";p.sid=req.sid;if(req.d)p.data=req.d; break;
    case "close":   p.op="close";p.sid=req.sid; break;
    default: return _json({e:"unknown op:"+req.t});
  }
  var r = UrlFetchApp.fetch(TUNNEL_SERVER_URL+"/tunnel",{
    method:"post",contentType:"application/json",
    payload:JSON.stringify(p),muteHttpExceptions:true,
  });
  if (r.getResponseCode()!==200) return _json({e:"tunnel HTTP "+r.getResponseCode()});
  return ContentService.createTextOutput(r.getContentText())
    .setMimeType(ContentService.MimeType.JSON);
}

function _doTunnelBatch(req) {
  var r = UrlFetchApp.fetch(TUNNEL_SERVER_URL+"/tunnel/batch",{
